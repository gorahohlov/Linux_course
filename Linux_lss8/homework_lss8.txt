# смотрю в чем дело
docker rm ubuntu1
# требуется запускать контейнер с запущенным приложением! Иначе он сразу останавливается.
docker run -it --name ubuntu1 ubuntu /bin/bash
# вошел в контейнер, приглашение изменилось.

# еще рабочий вариант 
# удаляю контейнер ubuntu2 созданный ранне в фоне - без запушенного приложения контейнер падает сразу после запуска и поэтому бесплезен.
docker rm ubuntu2
# создаю новый контейнер в интерактивном режиме
docker run -ti --name ubuntu2 -v /var/cntr_ubu2/:/usr/share/ubu2 ubuntu
# успешно приглашение поменялось, команды в выполняются в новом окружении;
# сразу после выхода контейнер останавливается
docker restart ubuntu2
docker exec -ti ubuntu2 bash
# ok, работает

docker exec -ti ubuntu1 /bin/bash
# ok, работает
